$ \int x dx = \frac{1}{2}x^2+C $

$ \displaystyle F^\prime(x)=f(x) \quad \text{then} \quad \int f(x)dx=F(x)+C $







$$
\begin{aligned}
\frac{d}{dx}(x^n) &= n x^{n-1} && \quad \int x^n \, dx = \frac{x^{n+1}}{n+1} + C \\\\[6pt]
\frac{d}{dx}(\sin x) &= \cos x && \quad \int \cos x \, dx = \sin x + C \\\\[6pt]
\frac{d}{dx}(\cos x) &= -\sin x && \quad \int \sin x \, dx = -\cos x + C \\\\[6pt]
\frac{d}{dx}(\ln x) &= \frac1x && \quad \int \frac1x \, dx = \ln|x| + C \\\\[6pt]
\frac{d}{dx}(e^x) &= e^x && \quad \int e^x \, dx = e^x + C \\\\[6pt]
\frac{d}{dx}(a^x) &= a^x\ln a && \quad \int a^x \, dx = \frac{a^x}{\ln a} + C
\end{aligned}
$$
